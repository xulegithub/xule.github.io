def download_file(url, local_filename):  
    try:  
        response = requests.get(url)  
        if response.status_code == 200:  
            with open(local_filename, 'wb') as f:  
                f.write(response.content)  
            return True  
        else:  
            print(f"Failed to download {url}: HTTP Status Code {response.status_code}")  
    except requests.exceptions.RequestException as e:  
        print(f"Error downloading file from {url}:", str(e))  
        return False, url  
  
def extract_process_description(html):  
    soup = BeautifulSoup(html, 'html.parser')  
    process_blocks = []  # List to hold extracted plaintext blocks for descriptions  
      
    for block in soup.find_all('div', class_='plaintext'):  
        try:  
            description = re.sub('<[^>]*>', '', str(block))  # Remove HTML tags if present  
            process_blocks.append(description)  
        except Exception as e:  
            print("Error extracting plain text from block:", str(e))  
    return '\n'.join(process_blocks)  
  
def parse_equipment_specs(filename):  
    try:  
        with open(filename, 'r') as f:  
            specs = json.load(f)

        return specs, missing_keys  
    except json.JSONDecodeError as e:

        raise
